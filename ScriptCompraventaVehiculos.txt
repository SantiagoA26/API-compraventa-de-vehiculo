create database Compraventa_vehiculos
use Compraventa_vehiculos

create table Vehiculo
(
Placa varchar(7) primary key,
Marca varchar(50) not null,
Modelo varchar (30) not null,
AñoVehiculo int not null,
Precio int not null,
Tipo varchar(10) not null,
Disponibilidad Char(2) not null
)

create table DetalleDeCompra
(
NumCompra int identity(10,1) primary key ,
Fecha smalldatetime not null,
NombreComprador varchar(50) not null,
CedulaComprador varchar(20) not null,
CelularComprador varchar(15) not null,
Placa varchar(7) not null,
Precio int not null,
FOREIGN KEY (Placa) REFERENCES Vehiculo(Placa) ON UPDATE CASCADE
)

Create trigger Disponibilidad
ON DetalleDeCompra
AFTER INSERT, DELETE
AS
BEGIN
   
    UPDATE Vehiculo
    SET Disponibilidad = 'No'
    WHERE Placa IN (SELECT Placa FROM inserted);

   
    UPDATE Vehiculo
    SET Disponibilidad = 'Si'
    WHERE Placa IN (SELECT Placa FROM deleted)
      AND Placa NOT IN (SELECT Placa FROM DetalleDeCompra);
END;

-- P.A para cambiar información de un vehiculo excepto la placa
Create Proc spUpdateVehicle
@Placa varchar(7),
@Marca varchar(50),
@Modelo varchar (30),
@AñoVehiculo int,
@Precio int,
@Tipo varchar(10),
@Disponibilidad Char(2)
As
Begin
update Vehiculo set Marca = @Marca, Modelo = @Modelo, AñoVehiculo = @AñoVehiculo, Precio = @Precio, Tipo = @Tipo, Disponibilidad = @Disponibilidad
Where Placa = @Placa
End
